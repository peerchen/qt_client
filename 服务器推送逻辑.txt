struct YlsEveveryParam
{
	int  nSize    ;
	int  nTotal   ;
	int* pChanged ;

	{
		nSize    = 0;
		nTotal   = 0;
		pChanged = NULL;
	}
	~YlsEveveryParam()
	{
		if( pChanged )
		{
			delete[] pChanged;
			pChanged = NULL;
		}
	}
};
